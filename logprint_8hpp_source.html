200struct CHex
201{
202 static std::string From(const unsigned char* ptr, size_t length, bool upper = false)
203 {
204 const char* HEX = upper ? "0123456789ABCDEF" : "0123456789abcdef";
205 if (!ptr || !length)
206 {
207 return "";
208 }
209 std::string buffer;
210 buffer.reserve(length * 3 + 1);
211 for (size_t i = 0; i < length; ++i)
212 {
213 buffer.push_back(HEX[(ptr[i] >> 4) & 0x0F]);
214 buffer.push_back(HEX[(ptr[i]) & 0x0F]);
215 buffer.push_back(' ');
216 }
217 buffer.pop_back();
218 return buffer;
219 }
220
221 static std::string FromEnd(const unsigned char* ptr, size_t length, bool upper = false)
222 {
223 unsigned char* reptr = new unsigned char[length];
224 for (size_t i = 0; i < length; ++i)
225 {
226 reptr[i] = ptr[length - i - 1];
227 }
228 std::string buffer = From(reptr, length, upper);
229 delete[] reptr;
230 return buffer;
231 }
232};

265template <typename T>
266std::string strbit(const T& val)
267{
268 static const char *BIT[16] = {
269 [ 0] = "0000", [ 1] = "0001", [ 2] = "0010", [ 3] = "0011",
270 [ 4] = "0100", [ 5] = "0101", [ 6] = "0110", [ 7] = "0111",
271 [ 8] = "1000", [ 9] = "1001", [10] = "1010", [11] = "1011",
272 [12] = "1100", [13] = "1101", [14] = "1110", [15] = "1111",
273 };
274
275 std::string buffer;
276 buffer.reserve(sizeof(T) * 9 + 1);
277 std::string hex = strhex(val);
278 for (auto c : hex)
279 {
280 if (c >= '0' && c <= '9')
281 {
282 buffer.append(BIT[c-'0']);
283 }
284 else if (c >= 'a' && c <= 'f')
285 {
286 buffer.append(BIT[c-'a'+10]);
287 }
288 else
289 {
290 buffer.push_back(c);
291 }
292 }
293
294 return buffer;
295}
